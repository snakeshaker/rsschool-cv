   function generateHashtag(str) {
      if (str.trim().length == 0) return false;
      let arr = str.split(/\s+/);
      for (let i = 0; i < arr.length; i++) {
         arr[i] = arr[i].charAt(0).toUpperCase() + arr[i].slice(1);
      }
      if (arr[0].length > 139) return false;
      return `#${arr.join("")}`;
   }
